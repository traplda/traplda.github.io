mat<-matrix(rep(0,80*80),nrow=80)
for (i in 1:nrow(unbiased)) {
  xi <- (unbiased[i,2]+25)*(80/55)
  yi <- (unbiased[i,3]+30)*(80/55)
  mat[xi,yi] <- mat[xi,yi] + 1
}

fes_unbiased<- (-8.314*0.001*300)*log(mat)
fes_unbiased[is.infinite(fes_unbiased)] <- 8.314*0.3*log(nrow(unbiased))
fes_unbiased <- fes_unbiased - min(fes_unbiased)
